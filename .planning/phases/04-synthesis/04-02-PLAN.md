    Export `findExistingArticle()` with this signature:

    ```typescript
    import type { Article } from '../types/article.js';
    import type { WikiStore } from '../store/wiki-store.js';
    import { buildIndex, search } from '../search/search-index.js';
    import { generateText } from '../llm/adapter.js';
    import { buildTiebreakPrompt } from './prompt-builder.js';
    import { parseTiebreakDecision } from './output-parser.js';

    export const BM25_DEDUP_THRESHOLD = 3.0;

    export async function findExistingArticle(
      plannedTitle: string,
      store: WikiStore,
      existingArticles: Article[],
    ): Promise<Article | null> {
      // Tier 1: exact slug match (per D-12 — use WikiStore.slugify)
      const slug = store.slugify(plannedTitle);
      const exact = await store.getArticle(slug);
      if (exact) return exact;

      // Tier 2: BM25 near-match
      if (existingArticles.length === 0) return null;
      const index = buildIndex(existingArticles);
      const results = search(index, plannedTitle);
      const topResult = results[0];
      if (!topResult || topResult.score < BM25_DEDUP_THRESHOLD) return null;

      // Tier 3: LLM tiebreak (per D-11)
      const candidate = await store.getArticle(topResult.slug);
      if (!candidate) return null;

      process.stderr.write(
        `  [DEDUP] BM25 match: "${topResult.slug}" (score ${topResult.score.toFixed(1)}) — asking LLM...\n`
      );

      const prompt = buildTiebreakPrompt(plannedTitle, candidate.frontmatter.summary);
      const response = await generateText(prompt, { temperature: 0 });
      const decision = parseTiebreakDecision(response);

      process.stderr.write(`  [DEDUP] LLM decision: ${decision}\n`);
      return decision === 'update' ? candidate : null;
    }
    ```

    Key design decisions:
    - `store` is passed as argument (not imported directly) for testability — tests mock WikiStore
    - `existingArticles` passed in to avoid redundant `listArticles()` calls (caller already has the list)
    - Only API errors propagate — parse errors from tiebreak default to 'new' (safe)
    - Progress logged to stderr per INTG-02 contract

    **1. Create src/synthesis/wikilink-sanitizer.ts (per D-07, RESEARCH Pattern 3):**

    Single pure function, no dependencies:

    ```typescript
    const WIKILINK_RE = /\[\[([^\]]+)\]\]/g;

    /**
     * Strip wikilinks not matching a known article slug.
     * Handles both [[slug]] and [[slug|display text]] formats.
     * Per D-07: zero hallucinated links — hard constraint.
     */
    export function stripHallucinatedWikilinks(body: string, knownSlugs: Set<string>): string {
      return body.replace(WIKILINK_RE, (match, inner: string) => {
        const parts = inner.split('|');
        const slug = parts[0].trim();
        if (knownSlugs.has(slug)) return match;
        // Strip the wikilink — use display text if present, otherwise slug
        return parts[1]?.trim() ?? slug;
      });
    }
    ```

    **2. Create src/synthesis/article-builder.ts (per D-05, D-14, D-15, D-16, RESEARCH Pattern 6):**

    Two exported functions that assemble Article objects from ParsedArticle data:

    ```typescript
    import slugifyLib from 'slugify';
    import type { Article, Frontmatter } from '../types/article.js';
    import type { ParsedArticle } from './types.js';
    import { stripHallucinatedWikilinks } from './wikilink-sanitizer.js';

    /**
     * Build a new Article from LLM-parsed output.
     * Per D-05: sources from SourceRef URLs, sourced_at = now.
     * Per D-15: type: 'web' always for synthesis-created articles.
     * Per D-07: wikilinks sanitized before returning.
     */
    export function buildNewArticle(
      parsed: ParsedArticle,
      knownSlugs: Set<string>,
    ): Article {
      const now = new Date().toISOString();
      const sourceUrls = parsed.sourceRefs.map(r => r.url);
      const sanitizedBody = stripHallucinatedWikilinks(parsed.body, knownSlugs);

      return {
        slug: slugifyLib(parsed.title, { lower: true, strict: true }),
        frontmatter: {
          title: parsed.title,
          tags: [],
          categories: parsed.categories.length > 0 ? parsed.categories : ['Uncategorized'],
          sources: sourceUrls,
          sourced_at: now,
          type: 'web',
          created_at: now,
          updated_at: now,
          summary: parsed.summary,
        },
        body: sanitizedBody,
      };
    }

    /**
     * Build an updated Article by merging new LLM output with existing article.
     * Per D-14: sources = union of old + new URLs; updated_at and sourced_at refreshed.
     * Per D-13: body comes from LLM (which received existing body + new sources).
     */
    export function buildUpdatedArticle(
      existingArticle: Article,
      parsed: ParsedArticle,
      knownSlugs: Set<string>,
    ): Article {
      const now = new Date().toISOString();
      const newSourceUrls = parsed.sourceRefs.map(r => r.url);
      const mergedSources = [...new Set([
        ...existingArticle.frontmatter.sources,
        ...newSourceUrls,
      ])];
      const sanitizedBody = stripHallucinatedWikilinks(parsed.body, knownSlugs);

      return {
        slug: existingArticle.slug,
        frontmatter: {
          ...existingArticle.frontmatter,
          categories: parsed.categories.length > 0 ? parsed.categories : existingArticle.frontmatter.categories,
          sources: mergedSources,
          sourced_at: now,
          updated_at: now,
          summary: parsed.summary,
          // created_at preserved from existing article
        },
        body: sanitizedBody,
      };
    }
    ```

    Key design decisions:
    - Uses `slugifyLib` directly (same library as WikiStore.slugify — per D-12 consistency)
    - Categories default to `['Uncategorized']` if LLM provides none (per RESEARCH Open Question 3)
    - Tags set to empty array (LLM tag generation deferred — not in phase requirements)
    - wikilink sanitization applied inside builder (guaranteed — caller can't skip it)
    - Per RESEARCH Anti-Pattern: frontmatter is constructed programmatically, NEVER from LLM output
